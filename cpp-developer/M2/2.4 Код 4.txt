#include <iostream>

int main() {
    int baseRate = 3;
    int square = 65;
    int residentsCount = 3;

    int total = baseRate * square * residentsCount;

    std::cout << "Стоимость отопления для площади " << square << " кв. м. и количества жильцов "
              << residentsCount << " составит: " << total << " руб.";
}
